        const { src, dest, watch, series, parallel } = require("gulp");
        const sass = require("gulp-sass")(require("sass"));
        const postcss = require("gulp-postcss");
        const autoprefixer = require("autoprefixer");
        const cssnano = require("cssnano");
        const sourcemaps = require("gulp-sourcemaps");
        const rename = require("gulp-rename");
        const concat = require("gulp-concat");
        const uglify = require("gulp-uglify");
        const browserSync = require("browser-sync").create();

        // パス（今の構成に合わせて）
        const paths = {
          scss: "scss/**/*.scss",
          js: "js/**/*.js",
          cssOut: "css",
          jsOut: "js",
        };

        // SCSS -> css/*.min.css
        function styles() {
          return src(paths.scss, { sourcemaps: true })
            .pipe(sourcemaps.init())
            .pipe(sass().on("error", sass.logError))
            .pipe(postcss([autoprefixer(), cssnano()]))
            .pipe(rename({ suffix: ".min" }))
            .pipe(sourcemaps.write("."))
            .pipe(dest(paths.cssOut))
            .pipe(browserSync.stream());
        }

        // JS -> js/bundle.min.js
        function scripts() {
          return src([paths.js, "!js/**/*.min.js", "!js/**/bundle.min.js"])
            .pipe(concat("bundle.min.js"))
            .pipe(uglify())
            .pipe(dest(paths.jsOut))
            // JSはstream注入じゃなく「リロード」でOK（安定）
            .on("end", browserSync.reload);
        }

        function serve() {
          browserSync.init({
            server: { baseDir: "./" }, // ← index.html がある場所
            open: true,                // ← まずは確実に開く
            notify: false,
          });

          watch(paths.scss, styles);

          // 生成物 bundle.min.js を監視から除外（無限ループ防止）
          watch(["js/**/*.js", "!js/**/bundle.min.js"], scripts);

          watch(["**/*.html"]).on("change", browserSync.reload);
        }

        exports.default = series(parallel(styles, scripts), serve);
